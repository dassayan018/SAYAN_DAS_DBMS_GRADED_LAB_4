SELECT 
  S.SUPP_ID, 
  S.SUPP_NAME, 
  S.SUPP_CITY, 
  S.SUPP_PHONE 
FROM 
  SUPPLIER S 
  INNER JOIN (
    SELECT 
      P.PRO_ID, 
      P.PRO_NAME, 
      SP.SUPP_ID 
    FROM 
      PRODUCT P 
      INNER JOIN SUPPLIER_PRICING SP ON P.PRO_ID = SP.PRO_ID 
    GROUP BY 
      SP.SUPP_ID 
    HAVING 
      COUNT(SP.SUPP_ID > 1)
  ) AS T1 ON T1.SUPP_ID = S.SUPP_ID;
