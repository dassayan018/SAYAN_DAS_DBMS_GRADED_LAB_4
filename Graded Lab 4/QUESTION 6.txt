SELECT 
  T1.CAT_ID, 
  T1.CAT_NAME, 
  SP.PRO_ID, 
  MIN(SP.SUPP_PRICE) 
FROM 
  SUPPLIER_PRICING SP 
  INNER JOIN (
    SELECT 
      C.CAT_ID, 
      C.CAT_NAME, 
      P.PRO_ID 
    FROM 
      CATEGORY C 
      INNER JOIN PRODUCT P ON C.CAT_ID = P.CAT_ID
  ) AS T1 ON T1.PRO_ID = SP.PRO_ID 
GROUP BY 
  T1.CAT_ID;
